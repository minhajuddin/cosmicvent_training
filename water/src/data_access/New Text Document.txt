

<?php
  class Artist{
    public $id;
    public $name;
    public $description;
    public $paintings;
    
    
    function Artist( $id, $name, $description,$paintings){
      $this->id = $id;
      $this->name = $name;
      $this->description = $description;
      $this->paintings = $paintings;
      
    }
  }
?>

<?php

  class Painting{
    public $id;
    public $subject;
    public $description;
    public $categoryId;
  
  
    function painting($id,$subject,$description,$categoryId) {
      $this->id = $id;
      $this->subject = $subject;
      $this->description = $description;
      $this->categoryId = $categoryId;
    }
  }
